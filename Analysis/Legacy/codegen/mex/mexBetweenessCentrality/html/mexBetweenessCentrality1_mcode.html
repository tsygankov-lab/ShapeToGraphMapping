  2function betweenness = mexBetweenessCentrality(V,E,color,vertFilterList,location,r2sV,s2rV,nonLeafVerts)
  3
  4% r2sV(vertFilterList) = 1:length(s2rV);
  5nonCycleFlag = 1;
  6
  7colorInd = 1;
  8edgeFilterList       = E.color(:) == color & V.borderRegions(E.origin(:,1),colorInd) == location;
  9% s2rE                 = find(edgeFilterList);
 10% r2sE                 = sparse(size(edgeFilterList));
 11% r2sE(edgeFilterList) = 1:length(s2rE);
 12%edgeSubGraph         = records.halfEdge(edgeFilterList,:);
 13
 14
 15
 16% fullGraphBinary = zeros(size(records.vorVert,1),1);
 17
 18validStartPoints = find(vertFilterList & nonLeafVerts);
 19betweenness = zeros(length(validStartPoints),1);
 20
 21
 22for vsp = 1:length(validStartPoints)
 23    vertexCollectionList = zeros(sum(vertFilterList),1);
 24    edgeCollectionList   = zeros(sum(edgeFilterList),1);
 25    
 26    edgeProcessed        = zeros(size(edgeFilterList),'uint8');
 27    vertexTerminal       = V.vertexCoorID(:) ~= -1;
 28    vertexTerminal_SubGraph = V.vertexCoorID(vertFilterList) ~= -1;
 29    vertexTerminal_SubGraph_Proc = vertexTerminal_SubGraph;
 30    vertexTerminal_SubGraph_Index = find(vertexTerminal_SubGraph);
 31    vertexTerminal_circList = zeros(size(vertexTerminal_SubGraph_Index));
 32    childEdge  = zeros(size(s2rV,1),2);
 33    parentEdge = zeros(size(s2rV,1),2); %%'2' because concave edges can have two parents
 34    radMatrix  = zeros(size(s2rV,1),1);
 35    distMatrix = zeros(size(s2rV,1),1);
 36    dirMat = zeros(size(s2rV,1),2);
 37    startPoint = validStartPoints(vsp);%s2rV(startPointID);%records.vorVert(s2rV(startPointID),:);
 38    startPointID = r2sV(startPoint);
 39    dirMat(1,:) = -[0,1];  %downward at start
 40    radMatrix(1) = V.rad(startPoint);%startPoint(3);
 41    distMatrix(1) = 0;
 42    vertexCollectionList(1) = startPoint;%s2rV(startPointID);
 43    parentEdge((startPointID),1) = -1;
 44    endState = 0;
 45    iter = 1;
 46    dbgplot = 0;
 47    edge = 0;
 48%     reversing = 0;
 49    terminals = 0;
 50    startingVertexInd = vertexCollectionList(1);
 51    %%loop until a termination condition has been met
 52    while(~endState)
 53        %     if(totalCount == 187)
 54        %         disp('pause')
 55        %     end
 56        curVert = vertexCollectionList(iter);%records.vorVert(vertexCollectionList(iter),:);
 57        if(~vertexTerminal(vertexCollectionList(iter))) %%If it is not a terminal vertex...
 58            if(parentEdge(r2sV(vertexCollectionList(iter)),2) == 0) %%Store the parent edge
 59                parEdge = parentEdge(r2sV(vertexCollectionList(iter)),1);
 60            else
 61                parEdge = parentEdge(r2sV(vertexCollectionList(iter)),2);
 62            end
 63            edgeIndexMatrixFiltConnected  = V.edgeIDList(curVert,1:V.degree(curVert));%curVert(9:8+curVert(8));		%%Edges connected to this vertex
 64            edgeIndexMatrixFiltInSubGraph = edgeIndexMatrixFiltConnected(edgeFilterList(edgeIndexMatrixFiltConnected));  %%Edges in subgraph region
 65            edgeIndexMatrixFiltNoReverse  = edgeIndexMatrixFiltInSubGraph((edgeIndexMatrixFiltInSubGraph) ~= parEdge); %%Edge is not an immediate reversal
 66            edgeIndexMatrixFinal          = edgeIndexMatrixFiltNoReverse(~edgeProcessed(edgeIndexMatrixFiltNoReverse)); %%Edge has not been touched
 67            
 68            if(isempty(edgeIndexMatrixFinal) && vertexCollectionList(iter) == startingVertexInd && nonCycleFlag)
 69                %%all paths exhausted, terminate...
 70                iter = iter + 1; %to not purge information...
 71                endState = 1;
 72                break;
 73            elseif(isempty(edgeIndexMatrixFinal)) %%If no edges are valid, reverse
 74                iter = iter - 1;
 75%                 if(dbgplot)
 76%                     plot(V.pos(curVert,1),V.pos(curVert,2),'rx','MarkerSize',6,'MarkerFaceColor','r');
 77%                     xlim([V.pos(curVert,1)-20,V.pos(curVert,1)+20]);
 78%                     ylim([V.pos(curVert,2)-20,V.pos(curVert,2)+20]);
 79%                     %                 f = getframe;
 80%                     totalCount = totalCount + 1;
 81%                     %                 imwrite(f.cdata,strcat('D:\Miracles\uglyAGAIN',num2str(totalCount), '.tiff'));
 82%                     drawnow;
 83%                 end
 84%                 reversing = 1;
 85            else
 86%                 if(dbgplot && ~reversing)
 87%                     plot(V.pos(curVert,1),V.pos(curVert,2),'go','MarkerSize',6,'MarkerFaceColor','g');
 88%                     xlim([V.pos(curVert,1)-20,V.pos(curVert,1)+20]);
 89%                     ylim([V.pos(curVert,2)-20,V.pos(curVert,2)+20]);
 90%                     totalCount = totalCount + 1;
 91%                     %                 f = getframe;
 92%                     %                 imwrite(f.cdata,strcat('D:\Miracles\uglyAGAIN',num2str(totalCount), '.tiff'));
 93%                     drawnow;
 94%                 end
 95%                 reversing = 0;
 96                if(length(edgeIndexMatrixFinal) == 1) %%Don't test if only one edge is there
 97                    edgeInd = edgeIndexMatrixFinal(1);
 98                else
 99                    angleMat = zeros(size(edgeIndexMatrixFinal));
100                    for i = 1:length(edgeIndexMatrixFinal)
101                        edge = edgeIndexMatrixFinal(i);%records.halfEdge(edgeIndexMatrixFinal(i),:);
102                        twinEdge = E.twinID(edge);%records.halfEdge(edge(1),:);
103                        if(sqrt(E.dir(edge,1)^2 + E.dir(edge,2)^2) < sqrt(eps))
104                            %This code branch is outdated, but there for safety
105                            noEdgeFound = 1;
106                            parEdge_T = E.twinID(edge);%edge(1); %%Twin edge, temporary parent edge for this cycle
107                            twinVert = E.origin(edge,2);%records.vorVert(edge(4),:);
108                            while(noEdgeFound)
109                                zeroedgeIndexMatrixFiltConnected  = V.edgeIDList(twinVert,1:V.degree(twinVert));%twinVert(9:8+twinVert(8));		%%Edges connected to this vertex
110                                zeroedgeIndexMatrixFiltInSubGraph = zeroedgeIndexMatrixFiltConnected(edgeFilterList(zeroedgeIndexMatrixFiltConnected));  %%Edges in subgraph region
111                                zeroedgeIndexMatrixFiltNoReverse  = zeroedgeIndexMatrixFiltInSubGraph((zeroedgeIndexMatrixFiltInSubGraph) ~= parEdge_T); %%Edge is not an immediate reversal
112                                zeroedgeIndexMatrixFinal          = zeroedgeIndexMatrixFiltNoReverse(~edgeProcessed(zeroedgeIndexMatrixFiltNoReverse)); %%Edge has not been touched
113                                if(isempty(zeroedgeIndexMatrixFinal))
114                                    %If this is true, this path might lead to
115                                    %a vertex? - Or on terminating iteration
116                                    noEdgeFound = 0;
117                                    angleMat(i) = -2; %%Only occurs on the terminating iteration...
118                                else
119                                    for j = 1:length(zeroedgeIndexMatrixFinal)
120                                        zeroEdge = zeroedgeIndexMatrixFinal(j);%records.halfEdge(zeroedgeIndexMatrixFinal(j),:);
121                                        if(sqrt(E.dir(zeroEdge,1)^2 + E.dir(zeroEdge,2)^2) > sqrt(eps(max([V.pos(E.origin(twinEdge,1),1:2)]))))
122                                            %%as long as the edge has a valid direction and its not a reversal it is fine.  The left most edge of one branch can not be to the left of the rightmost edge of the other if it is truly to the right
123                                            angleMat(i) = angleBetween(dirMat(iter,:),E.dir(zeroEdge,:));%zeroEdge(6:7)); %Alt - grab a linear dist calculation from the previous point
124                                            noEdgeFound = 0;
125                                        else
126                                            parEdge_T = E.twinID(zeroEdge);%zeroEdge(1);
127                                            twinVert = E.origin(zeroEdge,2);%records.vorVert(zeroEdge(4),:);
128                                        end
129                                    end
130                                end
131                            end
132                        else
133                            angleMat(i) = angleBetween(dirMat(iter,:),E.dir(edge,:));%edge(6:7));
134                        end
135                    end
136                    [~,ind] = min(angleMat); %%for CCW selection
137                    edgeInd = edgeIndexMatrixFinal(ind);
138                end
139                
140                edgeCollectionList(iter) = edgeInd;
141                edgeProcessed(edgeInd) = 1;
142                edge = edgeInd;%records.halfEdge(edgeInd,:);
143                iter = iter + 1;
144                
145                if(E.origin(edge,2) == startingVertexInd)
146                    %%TERMINATE LOOP - DO NOT STORE MORE INFORMATION
147                    endState = 1;
148                    break;
149                else
150                    vertexCollectionList(iter) = E.origin(edge,2);%(edge(4));
151                    subGraphIndOld = r2sV(vertexCollectionList(iter-1));
152                    subGraphIndNew = r2sV(vertexCollectionList(iter));
153                    if(parentEdge(subGraphIndNew,1) == 0)
154                        parentEdge(subGraphIndNew,1) = E.twinID(edge);%edge(1);
155                    else
156                        parentEdge(subGraphIndNew,2) = E.twinID(edge);%edge(1);
157                    end
158                    
159                    if(childEdge(subGraphIndOld,1) == 0)
160                        childEdge(subGraphIndOld,1) = edgeInd;
161                    else
162                        childEdge(subGraphIndOld,2) = edgeInd;
163                    end
164                    
165                    if(E.dist(edge) < sqrt(eps(max([V.pos(E.origin(edge,2),1:2)]))))
166                        dirMat(iter,:) = dirMat(iter-1,:);
167                    else
168                        dirMat(iter,:) = -E.dir(E.twinID(edge),:);%-records.halfEdge(edge(1),6:7);
169                    end
170                    
171                    distMatrix(iter) = E.dist(edge);%edge(5);
172                    radMatrix(iter) = V.rad(vertexCollectionList(iter));
173                end
174            end
175        else
176            if(vertexTerminal_SubGraph_Proc(r2sV(vertexCollectionList(iter))) == 1)
177                vertexTerminal_SubGraph_Proc(r2sV(vertexCollectionList(iter))) = 0; %Found
178                terminals = terminals + 1;
179                vertexTerminal_circList(terminals) = r2sV(vertexCollectionList(iter));
180%                 if(dbgplot)
181%                     plot(V.pos(curVert,1),V.pos(curVert,2),'go','MarkerSize',6,'MarkerFaceColor','g');
182%                     plot(V.pos(curVert,1),V.pos(curVert,2),'rx','MarkerSize',6,'MarkerFaceColor','r');
183%                     xlim([V.pos(curVert,1)-20,V.pos(curVert,1)+20]);
184%                     ylim([V.pos(curVert,2)-20,V.pos(curVert,2)+20]);
185%                     %                 f = getframe;
186%                     %                 imwrite(f.cdata,strcat('D:\Miracles\uglyAGAIN',num2str(totalCount), '.tiff'));
187%                     drawnow;
188%                 end
189            end
190            
191            reversing = 1;
192            iter = iter - 1;
193            
194        end
195    end
196    
197    
198    
199    vertexCollectionList(iter:end) = [];
200    % edgeCollectionList(iter:end) = [];
201%     dirMat(iter:end,:) = [];
202    % radMatrix(iter:end) = [];
203    % distMatrix(iter:end) = [];
204    % loopStruct.Vertices = vertexCollectionList;
205    % loopStruct.EL = edgeCollectionList;
206    % loopStruct.subGraphVerts = vertSubGraph;
207    % loopStruct.subGraphParentEdge = parentEdge;
208    % loopStruct.subGraphChildEdge = childEdge;
209    % loopStruct.direction = dirMat;
210    % loopStruct.incrDist = distMatrix;
211    % loopStruct.distance = cumsum(distMatrix);
212    % if(nonCycleFlag)
213    %     loopStruct.totalDistance = 0;
214    % else
215    %     loopStruct.totalDistance = loopStruct.distance(end) + E.dist(edgeCollectionList(end));%records.halfEdge(edgeCollectionList(end),5);
216    % end
217    % loopStruct.radius = radMatrix;
218    % loopStruct.records2SubgraphMap_Vertices = r2sV;
219    % loopStruct.subgraph2RecordsMap_Vertices = s2rV;
220    % loopStruct.binaryFilt = vertFilterList;
221    
222    % loopStruct.color = color;
223    % loopStruct.inObject = inObject;
224    % loopStruct.subGraphLeafFilter = vertexTerminal_SubGraph;
225    % loopStruct.nonCycle = nonCycleFlag;
226    isOnBridge = false(V.length,1);
227    isOnBridge(vertexCollectionList) = true;
228    distMat = zeros(size(vertexTerminal_circList));
229    vertTerminal_circList_rootNode = zeros(size(vertexTerminal_circList));
230    boundaryVertexCoor = zeros(size(vertTerminal_circList_rootNode,1),2);
231    for i = 1:length(vertexTerminal_circList)
232        %     if(i == 498)
233        %         disp('rev')
234        %     end
235        vertInd = s2rV(vertexTerminal_circList(i));
236        boundaryVertexCoor(i,:) = V.pos(vertInd,:);%records.vorVert(vertInd,1:2);
237        distTemp = 0;
238        if(~nonCycleFlag)
239            while(~isOnBridge(vertInd)) %should short circuit
240                edgeTraversal = parentEdge(r2sV(vertInd),1);%records.halfEdge(parentEdge(r2sV(vertInd),1),:);
241                distTemp = distTemp + E.dist(edgeTraversal);%edgeTraversal(5);
242                vertInd = E.origin(edgeTraversal,2);%edgeTraversal(4);
243            end
244        else
245            while(parentEdge(r2sV(vertInd),1) ~= -1)
246                edgeTraversal = parentEdge(r2sV(vertInd),1);%records.halfEdge(parentEdge(r2sV(vertInd),1),:);
247                distTemp = distTemp + E.dist(edgeTraversal);
248                vertInd = E.origin(edgeTraversal,2);
249            end
250        end
251        vertTerminal_circList_rootNode(i) = vertInd;
252        if(parentEdge(vertexTerminal_circList(i),2) ~= 0)
253            distTempB = 0;
254            vertInd = (vertexTerminal_circList(i));
255            edgeTraversal = parentEdge(vertInd,2);%parentEdge(vertInd,2)records.halfEdge(parentEdge(vertInd,2),:);
256            distTempB = distTempB + E.dist(edgeTraversal);%edgeTraversal(5);
257            vertInd = E.origin(edgeTraversal,2);%edgeTraversal(4);
258            if(~nonCycleFlag)
259                while(~isOnBridge(vertInd))
260                    edgeTraversal = parentEdge(r2sV(vertInd),1);%records.halfEdge(parentEdge(r2sV(vertInd),1),:);
261                    distTempB = distTempB + E.dist(edgeTraversal);%edgeTraversal(5);
262                    vertInd = E.origin(edgeTraversal,2);
263                end
264            else
265                while(parentEdge(r2sV(vertInd),1) ~= -1)
266                    edgeTraversal = parentEdge(r2sV(vertInd),1);%records.halfEdge(parentEdge(r2sV(vertInd),1),:);
267                    distTempB = distTempB + E.dist(edgeTraversal);
268                    vertInd = E.origin(edgeTraversal,2);
269                end
270            end
271            if(distTempB < distTemp)
272                distTemp = distTempB;
273                vertTerminal_circList_rootNode(i) = vertInd;
274            end
275        end
276        distMat(i) = distTemp;
277        if(distTemp - V.rad(vertTerminal_circList_rootNode(i)) < 0 && distTemp - V.rad(vertTerminal_circList_rootNode(i)) > -sqrt(eps))
278            distMat(i) = V.rad(vertTerminal_circList_rootNode(i));%records.vorVert(vertTerminal_circList_rootNode(i),3);
279            %         disp('Minor precision error in radius caught and accounted for...');
280        elseif(distTemp - V.rad(vertTerminal_circList_rootNode(i)) <= -sqrt(eps))
281            disp('Severe precision error detected, review loop plots');
282        end
283    end
284    betweenness(vsp) = sum(distMat);
285    % loopStruct.isRootNode = isOnBridge;
286    % loopStruct.parEdge = parentEdge;
287    % loopStruct.boundaryVerts = s2rV(vertexTerminal_circList);
288    % loopStruct.boundaryVertexCoor = boundaryVertexCoor; %Num = perim, Area requires calculation
289    % loopStruct.boundaryProfile = distMat;
290    % loopStruct.boundaryToCycleNode = vertTerminal_circList_rootNode;
291    % loopStruct.boundaryRatio = distMat./V.rad(vertTerminal_circList_rootNode);%records.vorVert(vertTerminal_circList_rootNode,3);
292    % loopStruct.boundaryDiff = distMat - V.rad(vertTerminal_circList_rootNode);
293end
