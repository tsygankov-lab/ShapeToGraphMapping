1function approx = angleBetween(dir1,dir2)
 2%%<2 - approaxing antiparallel
 3%%1 - 90 degree angle
 4%%0 - Parallel
 5%%-1 - 270 degree angle
 6%%-2 - antiparallel
 7%%assumes vectors are normalized already
 8mag1 = norm(dir1);
 9mag2 = norm(dir2);
10if(mag2 < 100*eps) %change to dynamic
11    approx = 0;
12else
13    dir1 = dir1./mag1;
14    dir2 = dir2./mag2;
15    sinth = dir1(1)*dir2(2) - dir1(2)*dir2(1); %%Orientation calculation
16    costh = dir1(1)*dir2(1) + dir1(2)*dir2(2); %%Dot product
17    
18    %%This calculates tan(theta) and pushes it closer to the extreme if the
19    %%angle is lower than 90 or above 270.  If you are not comfortable with
20    %%this, you may use the commented out version instead.
21    if(sinth == 0)
22        approx = sinth - (costh<eps)*(2*(1-abs(sinth)));
23    else
24        approx = sinth + (costh<eps)*(2*(1-abs(sinth)) * sign(sinth));
25    end
26end
